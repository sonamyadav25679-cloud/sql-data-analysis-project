CREATE TABLE users (
    user_id INT PRIMARY KEY,
    name VARCHAR(50),
    email VARCHAR(100)
);

CREATE TABLE jobs (
    job_id INT PRIMARY KEY,
    title VARCHAR(50),
    company VARCHAR(50),
    location VARCHAR(50)
);

CREATE TABLE applications (
    app_id INT PRIMARY KEY,
    user_id INT,
    job_id INT,
    status VARCHAR(20),
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (job_id) REFERENCES jobs(job_id)
);
